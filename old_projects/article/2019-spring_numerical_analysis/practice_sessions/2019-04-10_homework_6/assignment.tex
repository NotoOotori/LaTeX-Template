    \begin{problem}[三次样条插值]
        给定函数
        $\displaystyle f(x)=\frac{1}{1+25x^{2}}, -1\leq x\leq 1$,
        取等距节点, 构造牛顿插值多项式$N_{10}(x)$和三次样条插值函数
        $S_{10}(x)$. 分别将两种插值多项式与$f(x)$的曲线画在同一个坐标系
        上进行比较.
    \end{problem}

    \begin{solution}
        代码如下
        \lstinputlisting[
            caption={na6\_2.m},
            style=MATLAB-editor,
            basicstyle=\mlttfamily\scriptsize
        ]{na6_2.m}
        其中求解牛顿插值多项式的函数见之前的作业.

        得到图形如下
        \begin{figure}[H]
            \centering

    \end{solution}
